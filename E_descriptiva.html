6. Función para Crear Tablas Resumen de Análisis

Definimos una función que genera tablas resumen para cada variable (RSH, CONADI, PUEBLO). Esto reduce duplicación:

# Función corregida para crear tablas completamente horizontales
crear_tabla_resumen_horizontal <- function(data, var_label, etiqueta_variable) {
  
  # Paso 1: Calcular frecuencias y porcentajes sobre el total del año
  datos_calculados <- data %>%
    group_by(AÑO, GLOSA_SEXO, !!sym(var_label)) %>%
    summarise(Frecuencia = n(), .groups = "drop") %>%
    group_by(AÑO) %>% # Agrupamos solo por AÑO para que el % sea sobre el total anual
    mutate(
      # Crear la celda con el formato "Frecuencia (Porcentaje%)"
      ValorCelda = paste0(
        # CORREGIDO: Especificamos ambos separadores para evitar warnings
        format(Frecuencia, big.mark = ".", decimal.mark = ","), 
        " (", 
        scales::percent(Frecuencia / sum(Frecuencia), accuracy = 0.1), 
        ")"
      )
    ) %>%
    ungroup()

  # Paso 2: Pivotar usando la nueva columna 'ValorCelda'
  datos_calculados %>%
    # Seleccionamos solo las columnas necesarias para el pivot
    select(AÑO, GLOSA_SEXO, !!sym(var_label), ValorCelda) %>%
    pivot_wider(
      # La clave: combinar Sexo y la variable Label para los nombres de columna
      names_from = c(GLOSA_SEXO, !!sym(var_label)),
      values_from = ValorCelda,
      values_fill = "0 (0.0%)"
    ) %>%
    rename(Año = AÑO) # Renombramos AÑO y la tabla está lista
}

10. Función Reutilizable para Gráficos de Tendencia

Para evitar duplicar código, creamos una función que genera el gráfico de tendencia de “Pertenencia” para cualquier variable de interés (RSH, CONADI, Pueblo Originario).

# Versión mejorada con enfoque epidemiológico
crear_grafico_tendencia <- function(data, var_label, titulo_variable) {
    
  # Determinar el valor de "pertenencia" (numérico o texto)
  valor_pertenencia <- if (is.numeric(data[[var_label]])) {
    1
  } else {
    "Pertenece"
  }
  
  # Preparar los datos para el gráfico de prevalencia
  datos_tendencia <- data %>%
    filter(GLOSA_SEXO %in% c("HOMBRE", "MUJER")) %>%
    count(AÑO, GRUPO_ETARIO, GLOSA_SEXO, !!sym(var_label), name = "n") %>%
    group_by(AÑO, GRUPO_ETARIO, GLOSA_SEXO) %>%
    mutate(Porcentaje = n / sum(n)) %>%
    ungroup() %>%
    filter(!!sym(var_label) == valor_pertenencia) %>%
    # Asegurarnos de que AÑO sea numérico para el eje X
    mutate(AÑO = as.numeric(as.character(AÑO)))

  # Crear el gráfico de líneas con el AÑO en el eje X
  ggplot(datos_tendencia, aes(x = AÑO, y = Porcentaje, group = GLOSA_SEXO, color = GLOSA_SEXO)) +
    geom_line(linewidth = 1.2, alpha = 0.8) +
    geom_point(size = 3) +
    
    # --- CAMBIO CLAVE: Facetar por GRUPO_ETARIO ---
    facet_wrap(~ GRUPO_ETARIO, ncol = 3) +
    
    scale_y_continuous(labels = scales::percent_format(accuracy = 1)) +
    # Aseguramos que el eje X muestre los años como números enteros
    scale_x_continuous(breaks = unique(datos_tendencia$AÑO)) +
    scale_color_manual(values = c("HOMBRE" = "#F8766D", "MUJER" = "#00BFC4")) +
    
    # Títulos y etiquetas con terminología epidemiológica
    labs(
      title = paste("Evolución de la Prevalencia de Pertenencia a", titulo_variable),
      subtitle = "Tendencia anual por grupo de edad y sexo",
      x = "Año",
      y = "Prevalencia de Pertenencia",
      color = "Sexo"
    ) +
    theme_minimal(base_size = 12) +
    theme(
      axis.text.x = element_text(angle = 45, hjust = 1),
      legend.position = "bottom",
      panel.grid.minor = element_blank(),
      strip.text = element_text(face = "bold", size = 14)
    )
}
